# Environment ayarları
cp .env.example .env
# .env dosyasını düzenle

# Binary build
make build

# Çalıştır
./bin/server

# Systemd ile servis olarak çalıştır
sudo cp deployment/systemd/grpc-backend.service /etc/systemd/system/
sudo systemctl daemon-reload
sudo systemctl enable grpc-backend
sudo systemctl start grpc-backend
